CREATE DATABASE LAYOA;

CREATE TABLE TB_ROLE(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
	ROLE_CODE VARCHAR(20) COMMENT '角色名称',
	ROLE_NAME VARCHAR(20) COMMENT '角色名称',
	ROLE_KIND INT COMMENT '角色级别#1：超级角色；0：普通角色',
	ROLE_INFO VARCHAR(100) COMMENT '角色简介',
	ACTIVE_FLAG INT COMMENT'数据是否有效',
	CREATE_BY VARCHAR (10) comment'创建人',
	CREATE_DATE DATETIME comment'创建时间',
	UPDATE_BY VARCHAR(10) comment'修改人',
	UPDATE_DATE DATETIME comment'修改时间'	
)COMMENT'角色表';
